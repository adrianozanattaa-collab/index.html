    <!-- Firebase SDKs -->
    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
        import { getAuth, signInAnonymously, onAuthStateChanged, signInWithCustomToken } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
        import { getFirestore, doc, setDoc, onSnapshot, collection, query } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

        // Configuração do Ambiente
        const firebaseConfig = JSON.parse(__firebase_config);
        const app = initializeApp(firebaseConfig);
        const auth = getAuth(app);
        const db = getFirestore(app);
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'copa-2026-v1';

        let stickers = [];
        let userId = null;

        // Lista base para novos utilizadores
        const defaultStickers = [
            { id: 'FWC 00', name: 'Troféu Copa 2026', team: 'Especiais', owned: false, duplicates: 0 },
            { id: 'FWC 01', name: 'Logo Oficial', team: 'Especiais', owned: false, duplicates: 0 },
            { id: 'BRA 10', name: 'Neymar Jr.', team: 'Brasil', owned: false, duplicates: 0 },
            { id: 'ARG 10', name: 'Lionel Messi', team: 'Argentina', owned: false, duplicates: 0 },
            { id: 'POR 07', name: 'Cristiano Ronaldo', team: 'Portugal', owned: false, duplicates: 0 },
            { id: 'FRA 10', name: 'Kylian Mbappé', team: 'França', owned: false, duplicates: 0 },
            { id: 'USA 01', name: 'Matt Turner', team: 'EUA', owned: false, duplicates: 0 },
            { id: 'MEX 01', name: 'Guillermo Ochoa', team: 'México', owned: false, duplicates: 0 },
            { id: 'CAN 19', name: 'Alphonso Davies', team: 'Canadá', owned: false, duplicates: 0 }
        ];

        // Inicialização de Autenticação
        const initAuth = async () => {
            try {
                if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
                    await signInWithCustomToken(auth, __initial_auth_token);
                } else {
                    await signInAnonymously(auth);
                }
            } catch (error) {
                console.error("Erro auth:", error);
                document.body.innerHTML = `<div class="p-10 text-center text-red-600 font-bold">Erro de ligação. Por favor, recarregue a página.</div>`;
            }
        };

        onAuthStateChanged(auth, (user) => {
            if (user) {
                userId = user.uid;
                document.getElementById('user-id-display').textContent = `ID: ${userId}`;
                setupSync();
            }
        });

        // Sincronização Firestore (RULE 1 & 3)
        function setupSync() {
            if (!userId) return;

            const userDocRef = doc(db, 'artifacts', appId, 'users', userId, 'album', 'data');

            onSnapshot(userDocRef, (docSnap) => {
                if (docSnap.exists()) {
                    stickers = docSnap.data().list || [];
                } else {
                    // Novo utilizador: criar documento inicial
                    stickers = defaultStickers;
                    saveToCloud();
                }
                renderAll();
                document.getElementById('loading').classList.add('hidden');
            }, (error) => {
                console.error("Erro sync:", error);
            });
        }

        async function saveToCloud() {
            if (!userId) return;
            const userDocRef = doc(db, 'artifacts', appId, 'users', userId, 'album', 'data');
            try {
                await setDoc(userDocRef, { list: stickers });
            } catch (e) {
                console.error("Erro ao salvar:", e);
            }
        }

        // Funções de UI acessíveis globalmente
        window.toggleOwned = (id) => {
            const index = stickers.findIndex(s => s.id === id);
            stickers[index].owned = !stickers[index].owned;
            saveToCloud();
        };

        window.updateDuplicate = (id, amount) => {
            const index = stickers.findIndex(s => s.id === id);
            stickers[index].duplicates = Math.max(0, (stickers[index].duplicates || 0) + amount);
            saveToCloud();
        };

        window.copyUserId = () => {
            const tempInput = document.createElement("input");
            tempInput.value = userId;
            document.body.appendChild(tempInput);
            tempInput.select();
            document.execCommand("copy");
            document.body.removeChild(tempInput);
            alert("ID de acesso copiado! Guarde este ID para aceder aos seus dados noutro navegador.");
        };

        function renderAll() {
            renderDashboard();
            renderStickers();
        }

        function renderDashboard() {
            const total = stickers.length;
            const owned = stickers.filter(s => s.owned).length;
            const missing = total - owned;
            const duplicates = stickers.reduce((acc, s) => acc + (s.duplicates || 0), 0);
            const percentage = Math.round((owned / total) * 100) || 0;

            document.getElementById('stat-total').textContent = total;
            document.getElementById('stat-owned').textContent = owned;
            document.getElementById('stat-missing').textContent = missing;
            document.getElementById('stat-duplicates').textContent = duplicates;
            document.getElementById('progress-bar').style.width = percentage + '%';
            document.getElementById('progress-text').textContent = percentage + '% CONCLUÍDO';
        }

        function renderStickers() {
            const grid = document.getElementById('stickers-grid');
            const emptyState = document.getElementById('empty-state');
            const searchTerm = document.getElementById('search').value.toLowerCase();
            const filterValue = document.getElementById('filter-status').value;

            const filtered = stickers.filter(s => {
                const matchesSearch = s.name.toLowerCase().includes(searchTerm) || 
                                    s.id.toLowerCase().includes(searchTerm) || 
                                    s.team.toLowerCase().includes(searchTerm);
                
                let matchesFilter = true;
                if (filterValue === 'owned') matchesFilter = s.owned;
                if (filterValue === 'missing') matchesFilter = !s.owned;
                if (filterValue === 'duplicates') matchesFilter = s.duplicates > 0;

                return matchesSearch && matchesFilter;
            });

            grid.innerHTML = '';
            
            if (filtered.length === 0) {
                emptyState.classList.remove('hidden');
            } else {
                emptyState.classList.add('hidden');
                filtered.forEach(s => {
                    const card = document.createElement('div');
                    card.className = `sticker-card p-4 rounded-xl border-2 bg-white shadow-sm flex flex-col justify-between ${s.owned ? 'owned' : 'missing'}`;
                    
                    card.innerHTML = `
                        <div>
                            <div class="flex justify-between items-start mb-2">
                                <span class="text-xs font-bold text-gray-400">${s.id}</span>
                                <span class="text-[10px] bg-gray-200 px-2 py-0.5 rounded text-gray-600 uppercase font-bold">${s.team}</span>
                            </div>
                            <h3 class="font-bold text-gray-800 leading-tight mb-3">${s.name}</h3>
                        </div>
                        
                        <div class="space-y-3">
                            <button onclick="toggleOwned('${s.id}')" 
                                    class="w-full py-2 rounded-lg text-sm font-bold transition ${s.owned ? 'bg-green-500 text-white' : 'bg-gray-100 text-gray-600 border border-gray-300'}">
                                ${s.owned ? '✓ NO ÁLBUM' : 'MARCAR COLADA'}
                            </button>
                            
                            <div class="flex items-center justify-between bg-gray-50 p-1.5 rounded-lg border border-gray-200">
                                <span class="text-xs font-semibold text-gray-500 ml-2">Repetidas</span>
                                <div class="flex items-center gap-3">
                                    <button onclick="updateDuplicate('${s.id}', -1)" class="w-6 h-6 rounded bg-white border shadow-sm flex items-center justify-center font-bold text-gray-600 hover:bg-red-50">-</button>
                                    <span class="font-bold text-sm ${s.duplicates > 0 ? 'text-indigo-600' : 'text-gray-400'}">${s.duplicates}</span>
                                    <button onclick="updateDuplicate('${s.id}', 1)" class="w-6 h-6 rounded bg-white border shadow-sm flex items-center justify-center font-bold text-gray-600 hover:bg-green-50">+</button>
                                </div>
                            </div>
                        </div>
                    `;
                    grid.appendChild(card);
                });
            }
        }

        // Listeners básicos
        document.getElementById('search').addEventListener('input', renderStickers);
        document.getElementById('filter-status').addEventListener('change', renderStickers);

        // Start
        initAuth();

    </script>
